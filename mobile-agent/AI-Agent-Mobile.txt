// ── Common fetch headers ──────────────────────────
function headers(extra={}){

    ...(C.aiKey    ?{'X-AI-Key':C.aiKey}:{}),
    ...(C.aiUrl    ?{'X-AI-URL':C.aiUrl}:{}),
    ...(C.model    ?{'X-AI-Model':C.model}:{}),
    ...extra
  };
}

async function sendMsg(){
  const inp=q('#msg-input');
  const text=inp.value.trim();
  if(!text||generating) return;
  inp.value=''; autoH(inp);

  addUserMsg(text);
  chatHistory.push({role:'user',content:text});

  generating=true;
  q('#send-btn').style.display='none';
  q('#stop-btn').style.display='flex';

  const typId=addTyping();

  const msgs=[{role:'system',content:C.system},...chatHistory.slice(-28)];

  try{
    controller=new AbortController();
    const base=apiBase();

    let resp, fullText='';

    if(base){
      // ── Use backend (SSE endpoint) ──────────────────
      resp = await fetch(`${base}/api/ai/chat/stream`,{
        method:'POST',
        headers:headers(),
        body:JSON.stringify({messages:chatHistory.slice(-28),executeActions:true}),
        signal:controller.signal
      });

      if(!resp.ok) throw new Error('Backend HTTP '+resp.status+': '+await resp.text());

      removeTyping(typId);
      const mid=addAIMsg('',true);
      const reader=resp.body.getReader();
      const dec=new TextDecoder();
      let buf='';

      while(true){
        const {done,value}=await reader.read();
        if(done) break;
        buf+=dec.decode(value,{stream:true});
        const lines=buf.split('\n'); buf=lines.pop()||'';
        for(const line of lines){
          const t=line.trim();
          if(!t.startsWith('data:')) continue;
          const j=t.slice(5).trim();
          if(j==='[DONE]') continue;
          try{
            const p=JSON.parse(j);
            if(p.token){ fullText+=p.token; updateMsg(mid,fullText); }
            if(p.done)  { finalMsg(mid,p.fullText||fullText); }
            if(p.error) { finalMsg(mid,'❌ '+p.error); }
          }catch{}
        }
      }
      if(!fullText) fullText='_(Empty response)_';
      finalMsg(mid,fullText);

    }else{
      // ── Direct AI provider call (no backend) ────────
      const aiResp = await fetch(`${C.aiUrl}/chat/completions`,{
        method:'POST',
        headers:{
          'Content-Type':'application/json',
          ...(C.aiKey?{Authorization:`Bearer ${C.aiKey}`}:{}),
          ...(C.aiUrl.includes('openrouter')?{'HTTP-Referer':'https://vikramcpu.github.io','X-Title':'AI Coding Agent'}:{})
        },
        body:JSON.stringify({model:C.model,messages:msgs,max_tokens:4096,temperature:0.7,stream:true}),
        signal:controller.signal
      });

      if(!aiResp.ok) throw new Error('AI HTTP '+aiResp.status);
      removeTyping(typId);
      const mid=addAIMsg('',true);
      const reader=aiResp.body.getReader();
      const dec=new TextDecoder();
      let buf='';

      while(true){
        const {done,value}=await reader.read();
        if(done) break;
        buf+=dec.decode(value,{stream:true});
        const lines=buf.split('\n'); buf=lines.pop()||'';
        for(const line of lines){
          const t=line.trim();
          if(!t.startsWith('data:')) continue;
          const j=t.slice(5).trim();
          if(j==='[DONE]') continue;
          try{
            const p=JSON.parse(j);
            const tok=p.choices?.[0]?.delta?.content||'';
            if(tok){ fullText+=tok; updateMsg(mid,fullText); }
          }catch{}
        }
      }
      if(!fullText) fullText='_(Empty response)_';
      finalMsg(mid,fullText);
    }

    chatHistory.push({role:'assistant',content:fullText});
    displayHist.push({id:'m'+Date.now(),role:'assistant',content:fullText,time:timeNow()});
    saveHist();
    setDot('ok','Ready');

  }catch(e){
    removeTyping(typId);
    if(e.name==='AbortError') addAIMsg('⏹ _Stopped._');
    else{ setDot('err','Error'); addAIMsg(`❌ **Error:** ${e.message}\n\n_Check Settings (⚙️) — make sure Backend URL or AI Provider URL is correct._`); }
  }finally{
    generating=false; controller=null;
    q('#send-btn').style.display='flex';
    q('#stop-btn').style.display='none';
  }
}

function stopChat(){ controller?.abort(); }

// ── Message rendering ─────────────────────────────
function addUserMsg(text){
  const m={id:'m'+Date.now(),role:'user',content:text,time:timeNow()};
  displayHist.push(m); saveHist(); renderMsg(m,true); return m.id;
}
function addAIMsg(text,stream=false){
  const id='m'+Date.now()+'_ai';
  renderMsg({id,role:'assistant',content:text,time:timeNow()},true,stream);
  return id;
}
function renderMsg(m,scroll=true,stream=false){
  const isU=m.role==='user';
  const div=document.createElement('div');
  div.className='msg '+(isU?'user':'ai'); div.id=m.id;
  div.innerHTML=`
    <div class="av ${isU?'user':'ai'}">${isU?'👤':'🤖'}</div>
    <div class="bubble">
      <div class="bcontent">${isU?esc(m.content):md(m.content)}</div>
      ${stream?'':'<div class="btime">'+m.time+'</div>'}
    </div>`;
  q('#messages').appendChild(div);
  if(scroll) scrollBottom();
}
function updateMsg(id,text){
  const el=document.getElementById(id);
  if(el){ el.querySelector('.bcontent').innerHTML=md(text)+'<span style="opacity:.4">▋</span>'; scrollBottom(); }
}
function finalMsg(id,text){
  const el=document.getElementById(id);
  if(el){ el.querySelector('.bcontent').innerHTML=md(text); scrollBottom(); }
}
function addTyping(){
  const id='t'+Date.now();
  const div=document.createElement('div');
  div.className='msg ai'; div.id=id;
  div.innerHTML=`<div class="av ai">🤖</div><div class="bubble"><div class="typing"><span></span><span></span><span></span></div></div>`;
  q('#messages').appendChild(div); scrollBottom(); return id;
}
function removeTyping(id){ document.getElementById(id)?.remove(); }
function scrollBottom(){ const m=q('#messages'); if(m) m.scrollTop=m.scrollHeight; }
function clearChat(){
  chatHistory=[]; displayHist=[];
  q('#messages').innerHTML='';
  localStorage.removeItem(HIST_KEY);
  addAIMsg('Chat cleared! How can I help you? 🚀');
}

// ── Auto-fix Loop ─────────────────────────────────
let fixES=null;
function log(msg,cls='log-checking'){
  const box=q('#fix-log');
  const line=document.createElement('div');
  line.className='log-line '+cls;
  line.innerHTML=`<span class="lt">${timeNow()}</span><span>${msg}</span>`;
  box.appendChild(line);
  box.scrollTop=box.scrollHeight;
}

async function startAutoFix(){
  const owner=q('#fix-owner').value.trim();
  const repo=q('#fix-repo').value.trim();
  const workflow=q('#fix-workflow').value.trim();
  const branch=q('#fix-branch').value.trim()||'main';
  const max=q('#fix-max').value||5;
  const base=apiBase();

  if(!base){ toast('❌ Set Backend URL in ⚙️ Settings'); return; }
  if(!owner||!repo){ toast('❌ Fill Owner and Repo'); return; }

  q('#fix-log').innerHTML='';
  q('#fix-btn').disabled=true;
  q('#fix-stop').style.display='inline-flex';

  const url=`${base}/api/build/auto-fix?owner=${encodeURIComponent(owner)}&repo=${encodeURIComponent(repo)}&workflow=${encodeURIComponent(workflow)}&branch=${encodeURIComponent(branch)}&max_attempts=${max}`;
  fixES=new EventSource(url);

  fixES.addEventListener('status',e=>{
    const d=JSON.parse(e.data);
    const cls={
      success:'log-success',failed:'log-error',max_retries:'log-error',
      analyzing:'log-fixing',fixing:'log-fixing',committing:'log-commit',
      triggering:'log-info',waiting:'log-checking',checking:'log-checking',
      fetching_logs:'log-fixing'
    }[d.stage]||'log-checking';
    log(d.message,cls);
  });

  fixES.addEventListener('done',e=>{
    const d=JSON.parse(e.data);
    log(d.success?'🎉 BUILD SUCCEEDED!':'❌ Max attempts reached. Check GitHub manually.',d.success?'log-success':'log-error');
    stopAutoFix();
    if(d.success) toast('🎉 Build succeeded!');
    else toast('❌ Build failed after all attempts');
  });

  fixES.onerror=()=>{ log('❌ Connection error','log-error'); stopAutoFix(); };
}

function stopAutoFix(){
  fixES?.close(); fixES=null;
  q('#fix-btn').disabled=false;
  q('#fix-stop').style.display='none';
}

let reposLoaded=false;

async function loadRepos(){
  const base=apiBase();
  const el=q('#repos-list');
  if(!base&&!C.ghToken){ el.innerHTML=`<div class="empty-state"><div class="eico">⚙️</div><p>Set Backend URL or GitHub Token in Settings</p></div>`; return; }

  el.innerHTML=`<div class="empty-state"><div class="spin" style="font-size:24px">⏳</div><p style="margin-top:8px">Loading repos…</p></div>`;
  try{
    let repos;
    if(base){
      const r=await fetch(`${base}/api/github/repos`,{headers:headers()});
      const d=await r.json();
      if(!d.success) throw new Error(d.error);
      repos=d.data;
    }else{
      // Direct GitHub API
      const r=await fetch('https://api.github.com/user/repos?sort=updated&per_page=30',{
        headers:{Authorization:`token ${C.ghToken}`,'Accept':'application/vnd.github.v3+json'}
      });
      if(!r.ok) throw new Error('GitHub API '+r.status);
      repos=await r.json();
    }
    renderRepos(repos);
    reposLoaded=true;
  }catch(e){
    el.innerHTML=`<div class="empty-state"><div class="eico">❌</div><p>${e.message}</p></div>`;
  }
}

function renderRepos(repos){
  const el=q('#repos-list');
  if(!repos||repos.length===0){ el.innerHTML=`<div class="empty-state"><div class="eico">🗂️</div><p>No repos found</p></div>`; return; }
  el.innerHTML=repos.map(r=>`
    <div class="repo-item">
      <div class="repo-name">
        <a href="${r.html_url}" target="_blank" style="color:var(--brand2);text-decoration:none">${r.name} ↗</a>
      </div>
      <div class="repo-desc">${r.description||'No description'}</div>
      <div class="repo-meta">
        ${r.language?`<span class="tag lang">${r.language}</span>`:''}
        <span class="tag ${r.private?'priv':'pub'}">${r.private?'🔒 Private':'🌍 Public'}</span>
        <span class="tag">⭐ ${r.stargazers_count}</span>
        <span class="tag">🍴 ${r.forks_count}</span>
        <span class="tag" style="font-size:10px">${timeAgo(r.updated_at)}</span>
      </div>
    </div>`).join('');
}

function showCreateRepo(){ q('#create-repo-form').style.display='block'; q('#new-repo-name').focus(); }
function hideCreateRepo(){ q('#create-repo-form').style.display='none'; }

async function createRepo(){
  const name=q('#new-repo-name').value.trim();
  if(!name){ toast('❌ Enter a repo name'); return; }
  const base=apiBase();
  const desc=q('#new-repo-desc').value.trim();
  const priv=q('#new-repo-private').checked;
  const init=q('#new-repo-init').checked;

  try{
    let repo;
    if(base){
      const r=await fetch(`${base}/api/github/repos`,{method:'POST',headers:headers(),body:JSON.stringify({name,description:desc,private:priv,auto_init:init})});
      const d=await r.json(); if(!d.success) throw new Error(d.error);
      repo=d.data;
    }else{
      const r=await fetch('https://api.github.com/user/repos',{
        method:'POST',
        headers:{Authorization:`token ${C.ghToken}`,'Content-Type':'application/json','Accept':'application/vnd.github.v3+json'},
        body:JSON.stringify({name,description:desc,private:priv,auto_init:init})
      });
      if(!r.ok) throw new Error('GitHub '+r.status);
      repo=await r.json();
    }
    toast('✅ Repo "'+repo.name+'" created!');
    hideCreateRepo();
    loadRepos();
  }catch(e){ toast('❌ '+e.message); }
}

// MARKDOWN (lightweight)
// ════════════════════════════════════════════════
function md(t){
  if(!t) return '';
  let h=esc(t);
  h=h.replace(/```(\w*)\n?([\s\S]*?)```/g,(_,lang,code)=>{
    const id='c'+Math.random().toString(36).slice(2);
    return `<pre><button class="copy-code" onclick="cpCode('${id}')">Copy</button><code id="${id}">${code.trim()}</code></pre>`;
  });
  h=h.replace(/`([^`\n]+)`/g,'<code>$1</code>');
  h=h.replace(/^### (.+)$/gm,'<h3>$1</h3>');
  h=h.replace(/^## (.+)$/gm,'<h2>$1</h2>');
  h=h.replace(/^# (.+)$/gm,'<h1>$1</h1>');
  h=h.replace(/\*\*(.+?)\*\*/g,'<strong>$1</strong>');
  h=h.replace(/__(.+?)__/g,'<strong>$1</strong>');
  h=h.replace(/\*(.+?)\*/g,'<em>$1</em>');
  h=h.replace(/^[*\-•] (.+)$/gm,'<li>$1</li>');
  h=h.replace(/(<li>[\s\S]*?<\/li>)(\n(?!<li>)|$)/g,'<ul>$1</ul>');
  h=h.replace(/^\d+\. (.+)$/gm,'<li>$1</li>');
  h=h.replace(/\[([^\]]+)\]\((https?:\/\/[^\)]+)\)/g,'<a href="$2" target="_blank">$1</a>');
  h=h.replace(/\n\n+/g,'</p><p>');
  h=h.replace(/\n/g,'<br>');
  if(!h.startsWith('<')) h='<p>'+h+'</p>';
  return h;
}
function esc(t){ return t.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }
function cpCode(id){ navigator.clipboard?.writeText(document.getElementById(id)?.innerText||'').then(()=>toast('Copied!')); }

function timeAgo(iso){
  const s=Math.round((Date.now()-new Date(iso))/1000);
  if(s<60) return 'just now';
  if(s<3600) return Math.floor(s/60)+'m ago';
  if(s<86400) return Math.floor(s/3600)+'h ago';
  return Math.floor(s/86400)+'d ago';
}
